package Der;
import java.io.Serializable;
import java.util.ArrayList;
import java.util.Comparator;
import java.util.List;

import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;

import Obligations.*;

public class Derivative implements Serializable{
    private static final Logger logger = LogManager.getLogger(Derivative.class);
    private List<InsuranceObligation> obligations; 

    public Derivative() {
        this.obligations = new ArrayList<>();
        logger.info("Новий об'єкт Derivative створений.");
    }

    public void addObligation(InsuranceObligation obligation) {
        obligations.add(obligation);
        logger.info("Додано нове зобов’язання: " + obligation);
    }

    public void removeObligation(InsuranceObligation obligation) {
        obligations.remove(obligation);
        logger.info("Зобов’язання видалено: " + obligation);
    }

    public double calculateTotalValue() {
        double totalValue = 0;
        for (InsuranceObligation obligation : obligations) {
            totalValue += obligation.calculateValue();
        }
        logger.info("Розрахована загальна вартість деривативу: " + totalValue);
        return totalValue;
    }

    public void sortObligationsByRisk() {
        obligations.sort(Comparator.comparingDouble(InsuranceObligation::getRiskLevel));
        logger.info("Зобов’язання відсортовані за рівнем ризику.");
    }

    public List<InsuranceObligation> findObligations(double minRisk, double maxRisk, double minAmount, double maxAmount) {
        List<InsuranceObligation> filteredObligations = new ArrayList<>();
        for (InsuranceObligation obligation : obligations) {
            if (obligation.getRiskLevel() >= minRisk && obligation.getRiskLevel() <= maxRisk &&
                obligation.getAmount() >= minAmount && obligation.getAmount() <= maxAmount) {
                filteredObligations.add(obligation);
            }
        }
        logger.info("Знайдено зобов’язання за заданими параметрами: " + filteredObligations.size() + " зобов’язань.");
        return filteredObligations;
    }

    public List<InsuranceObligation> getObligations() {
        List<InsuranceObligation> filteredObligations = new ArrayList<>();
        for (InsuranceObligation obligation : obligations) {
            filteredObligations.add(obligation);
        }
        logger.info("Отримано список усіх зобов’язань. Кількість зобов’язань: " + obligations.size());
        return filteredObligations;
    }

    // Метод для виведення списку зобов’язань
    public void displayObligations() {
        System.out.println("Облiгацiї:");
        for (InsuranceObligation obligation : obligations) {
            System.out.println(obligation.toString());
        }
    }
}
